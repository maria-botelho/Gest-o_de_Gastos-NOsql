  { "$match": { "usuario_id": ObjectId("..."), "data": { "$gte": ISODate("{{ ano }}-{{ '%02d'|format(mes) }}-01") } } },

  { "$lookup": {

  { "$unwind": "$dados_conta" },

      "conta_tipo": "$dados_conta.tipo"
